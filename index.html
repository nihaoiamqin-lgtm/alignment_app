import gradio as gr def local_score_table(str1: str, str2: str, match: float, mismatch: float, gap: float) -> list[list[float]]: rows = len(str1) + 1 cols = len(str2) + 1 matrix = [[0.0] * cols for _ in range(rows)] for i in range(1, rows): for j in range(1, cols): if str1[i-1] == str2[j-1]: diag = matrix[i-1][j-1] + match else: diag = matrix[i-1][j-1] - mismatch up = matrix[i-1][j] - gap left = matrix[i][j-1] - gap matrix[i][j] = max(0.0, diag, up, left) return matrix def local_alignment(str1: str, str2: str, match: float, mismatch: float, gap: float): matrix = local_score_table(str1, str2, match, mismatch, gap) max_score = -1.0 r, c = 0, 0 for i in range(len(matrix)): for j in range(len(matrix[0])): if matrix[i][j] > max_score: max_score = matrix[i][j] r, c = i, j i, j = r, c align1, align2 = "", "" while matrix[i][j] > 0: CURRENT = matrix[i][j] LEFT = matrix[i][j-1] - gap UP = matrix[i-1][j] - gap if CURRENT == LEFT: align1 = "-" + align1 align2 = str2[j-1] + align2 j -= 1 elif CURRENT == UP: align1 = str1[i-1] + align1 align2 = "-" + align2 i -= 1 else: align1 = str1[i-1] + align1 align2 = str2[j-1] + align2 i -= 1 j -= 1 return f"Score: {max_score}\n\n{align1}\n{align2}" def global_alignment(str1: str, str2: str, match: float, mismatch: float, gap: float): rows = len(str1) + 1 cols = len(str2) + 1 matrix = [[0.0] * cols for _ in range(rows)] for i in range(1, rows): matrix[i][0] = matrix[i-1][0] - gap for j in range(1, cols): matrix[0][j] = matrix[0][j-1] - gap for i in range(1, rows): for j in range(1, cols): diag = matrix[i-1][j-1] + (match if str1[i-1] == str2[j-1] else -mismatch) up = matrix[i-1][j] - gap left = matrix[i][j-1] - gap matrix[i][j] = max(diag, up, left) i, j = rows - 1, cols - 1 align1, align2 = "", "" final_score = matrix[i][j] while i > 0 or j > 0: CURRENT = matrix[i][j] if i > 0 and j > 0 and CURRENT == matrix[i-1][j-1] + (match if str1[i-1] == str2[j-1] else -mismatch): align1 = str1[i-1] + align1 align2 = str2[j-1] + align2 i -= 1 j -= 1 elif i > 0 and CURRENT == matrix[i-1][j] - gap: align1 = str1[i-1] + align1 align2 = "-" + align2 i -= 1 else: align1 = "-" + align1 align2 = str2[j-1] + align2 j -= 1 return f"Score: {final_score}\n\n{align1}\n{align2}" def run_app(algo_type, seq1, seq2, match, mismatch, gap): if algo_type == "Local Alignment": return local_alignment(seq1.upper(), seq2.upper(), match, mismatch, gap) else: return global_alignment(seq1.upper(), seq2.upper(), match, mismatch, gap) app = gr.Interface( fn=run_app, inputs=[ gr.Radio(["Local Alignment", "Global Alignment"], label="Algorithm", value="Local Alignment"), gr.Textbox(label="Sequence 1", value="GATTACA"), gr.Textbox(label="Sequence 2", value="GCATGCU"), gr.Number(label="Match Score", value=1.0), gr.Number(label="Mismatch Penalty (Positive Number)", value=1.0), gr.Number(label="Gap Penalty (Positive Number)", value=1.0) ], outputs=gr.Textbox(label="Alignment Result", lines=5), title="Sequence Alignment Explorer", description="Explore how scoring choices change global and local alignments." ) app.launch()
